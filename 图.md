## 数组（邻接矩阵）表示法  
*缺点：对稀疏图而言浪费空间*  
<img width="469" height="288" alt="image" src="https://github.com/user-attachments/assets/d6ef01c8-8e30-4d6d-aae2-ddc3e32289a2" />  

    #include<stdio.h>

    typedef char VertexType;  //顶点类型，假设顶点的数据类型为字符型
    typedef int ArcType;      //假设边的权值类型为整型
    #define MAXSIZE 100       //最大顶点数

    //图的邻接矩阵结构体定义
    typedef struct{
        VertexType vertex[MAXSIZE];      //顶点数组
        ArcType arc[MAXSIZE][MAXSIZE];   //邻接矩阵（二维数组）
        int vertex_num;                   //顶点个数
        int arc_num;                     // 边的条数
    }Mat_Graph;                           // Matrix 矩阵

    //创建图的邻接矩阵
    void create_graph(Mat_Graph* G){
        G->vertex_num = 4;  //顶点V0~V3
        G->arc_num = 5;

        //顶点命名
        G->vertex[0] = '0';
        G->vertex[1] = '1';
        G->vertex[2] = '2';
        G->vertex[3] = '3';

        //初始化邻接矩阵
        for(int i = 0;i < G->vertex_num;i++){
            for(int j = 0;j < G->vertex_num;j++){
                G->arc[i][j] = 0;
            }
        }

        //添加边（无向图，因此要对称赋值;有向图则不是）
        G->arc[0][1] = 1;    G->arc[1][0] = 1;
        G->arc[0][2] = 1;    G->arc[2][0] = 1;
        G->arc[0][3] = 1;    G->arc[3][0] = 1;
        G->arc[1][2] = 1;    G->arc[2][1] = 1;
        G->arc[2][3] = 1;    G->arc[3][2] = 1;
    }

    //输出邻接矩阵
    void print_matrix(Mat_Graph G){
        printf("邻接矩阵表示如下：\n");
        for(int i = 0;i < G.vertex_num; i++){
            for(int j = 0;j < G.vertex_num;j++){
                printf("%d ",G.arc[i][j]);
            }
            printf("\n");
        }
    }

    int main(int argc, char const *argv[]){
        Mat_Graph G;  //创建一个邻接矩阵的结构体变量
        create_graph(&G);  //初始化邻接矩阵
        print_matrix(G);   //输出邻接矩阵
        reutrn 0;
    }

与上边代码的区别：  
1.最初定义时添加语句： #define MaxInt 32767  //或用0x7fffffff,int类型的最大值，表示极大值，即无穷    
2.初始化邻接矩阵时，默认全部为Max  
3.添加边时赋值为边权值大小  
4.输出时，如果该元素值为Max，让其输出字符串Max,而不是输出Max的值，如下：
~~~
                for(int i = 0;i < G.vertex_num; i++){
                    for(int j = 0;j < G.vertex_num;j++){
                        if(G.arc[i][j] == Max)
                            printf("Max ");
                        else
                            printf("%3d ",G.arc[i][j]);
                    }
                    printf("\n");
               }
~~~
