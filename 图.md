        G->vertex_num = 4;  //顶点V0~V3

        G->vertex[1] = '1';
        G->vertex[2] = '2';
        G->vertex[3] = '3';

        //初始化邻接矩阵
        for(int i = 0;i < G->vertex_num;i++){
            for(int j = 0;j < G->vertex_num;j++){
                G->arc[i][j] = 0;
            }
        }

        //添加边（无向图，因此要对称赋值;有向图则不是）

        G->arc[0][3] = 1;    G->arc[3][0] = 1;
        G->arc[1][2] = 1;    G->arc[2][1] = 1;
        G->arc[2][3] = 1;    G->arc[3][2] = 1;
    }

    void print_matrix(Mat_Graph G){
        printf("邻接矩阵表示如下：\n");

## 邻接表（链式）表示法
<img width="502" height="335" alt="image" src="https://github.com/user-attachments/assets/6cb6ddee-077c-475b-8365-8e6e13951080" />  

<img width="255" height="179" alt="image" src="https://github.com/user-attachments/assets/2ce8022d-fae3-4038-930c-2e44a4ab757b" />

*空间效率：e为边数，无向图中每个边计算两次再加上头结点的存储所以为O(n+2e)*

    #include<stdio.h>
    #include<stdlib.h>
    #define MAXVEX 100  //最大顶点数

    //定义边表结点结构体
    typedef struct ArcNode{
        int adjvex;            //邻接点在顶点数组中的下标
        struct ArcNode *next;  //指向下一个邻接点
    }ArcNode;

    //定义顶点表结点结构体
    typedef struct VertexNode{
        char data;            //顶点信息(这里用字符代替)
        ArcNode *firstarc;    //指向边表的头指针
    }VertexNode,AdjList[MAXVEX];  //AdjList[MAXVEX]等价于struct VertexNode AdjList[MAXVEX],也就是长度为MAXVEX的VertexNode结构体数组    

    //图的结构体
    typedef struct{
        AdjList adjList;  //邻接表    此处 AdjList adjList等价于VertexNode adjList[MAXVEX]
        int numVertexes;  //顶点数
        int numArcs;      //边数
    }GraphAdjList;

    //创建图 邻接表（无向图）
    void createGraph(GraphAdjList *G){
        G->numVertexes = 4;
        G->numArcs = 5;

        //顶点命名
        for(int i = 0;i < G->numVertexes;i++){
            G->adjList[i].data = '0' + i;    //这里相当于直接输入各顶点字符0，1，2，3，4，5；如果值不是这样，也可以直接进行输入
            G->adjList[i].firstarc = NULL;
        }

        //边集合
        int arcs[5][2] = {{0,1},{0,2},{0,3},{1,2},{2,3}};
        for(int i = 0;i < G->numArcs;i++){
            int from = arcs[i][0];
            int to = arcs[i][1];

            //from → to
            ArcNode *e1 = (ArcNode *)malloc(sizeof(ArcNode));
            e1->adjvex = to;
            e1->next = G->adjList[from].firstarc;  //此时firstarc为空值
            G->adjList[from].firstarc = e1;   //头插法

            //to → from（因为是无向图）
            ArcNode *e2 = (ArcNode *)malloc(sizeof(ArcNode));
            e2->adjvex = from;
            e2->next = G->adjList[to].firstarc;  
            G->adjList[to].firstarc = e2;
        }
    }

    //打印邻接表
    void printGraph(GraphAdjList G){
        printf("图的邻接表如下：\n");
        for(int i = 0;i < G.numVertexes;i++){
            printf("V%c -> ",G.adjList[i].data);   //打印的是0,1这些字符所以手动加上了个V
            ArcNode *p = G.adjList[i].firstarc;
            while(p){
                printf("%d -> ",p->adjvex);
                p = p->next;
            }
            printf("^\n");       //^表示空值
        }
    }

    int main(){
        GraphAdjList G;
        createGraph(&G);
        printfGraph(G);
        return 0;
    }

    void prim(Mat_Graph* G){
        int i,j,k;
        int min;
        int weight[MAXSIZE];    //候选边
        int vex_index[MAXSIZE];  //值表示出发点，下标表示到达点

        //先从第一个顶点开始
        weight[0] = 0;  //vex_index某点与下标对应点的权重
        vex_index[0] = 0;

        for(i = 1;i < G->vertex_num;i++){
            weight[i] = G->arc[0][i];  //把第0行对应权值赋给weight数组,所以就是在weight数组中去找下一条边故称之为候选边
            vex_index[i] = 0;
        }

        for(int i = 1;i < G->vertex_num;i++){
            min = MAX;  //min最初默认为最大值
            j = 0;
            k = 0;
            //找到最小的，与之连接
            while(j < G->vertex_num){
                if(weight[j] < min){
                    min = weight[j];
                    k = j;
                }
                j++;
            }

            printf("(%c,%c)\n",G->vertex[vex_index[k]],G->vertex[k]);
            weight[k] = 0;

            //找新连接结点的待选路径
            for(j = 0;j < G->vertex_num;j++){
                //与新结点可连接的边进行比较，有比weight小的就替换
                if(weight[j] != 0 && G->arc[k][j] < weight[j]){
                    weight[j] = G-arc[k][j];
                    vex_index[j] = k;
                }
            }
        }

    int choose(int distance[],int found[],int vertex_num){           //distance数组即上边图片中的D[]，记录距离;found数组即S[]
        int min = MAX;
        int minPos = -1;
        for(int i = 0;i < vertex_num;i++){
            if(distance[i] < min && found[i] == 0){
                min = distance[i];
                minPos = i;
            }
        }
        return minPos;
    }

    void dijkstra(Mat_Graph G,int begin){
        int found[MAXSIZE];      //顶点是否已走过
        int path[MAXSIZE];       //路径
        int distance[MAXSIZE];   //begin到达每一个顶点的最短距离
        for(int i = 0;i < G.vertex_num;i++){
            found[i] = 0;
            path[i] = -1;
            distance[i] = G.arc[begin][i];        //对各个数组初始化
        }

        found[begin] = 1;
        distance[begin] = 0;          //起始点到起始点的距离为0

        int next;                     //下一个要观察的顶点
        for(int i = 1;i < G.vertex_num;i++){
            next = choose(distance,found,G.vertex_num);
            found[next] = 1;
            for(int j = 0;j < G.vertex_num;j++){
                if(found[j] == 0){
                    if(distance[next] + G.arc[next][j] < distance[j]){
                        distance[j] = distance[next] + G.arc[next][j];
                        path[j] = next;
                    }
                }
            }
        }

        //输出
        for(int i = 0;i < G.vertex_num;i++){
            printf("V0 -> V%c : %d\n",vertex[i],distance[i]);

            int j = i;
            printf("V%c <- ",vertex[i]);
            while(path[j] != -1){
                printf("V%c <- ",vertex[j]);
                j = path[j];
            }
            printf("V0\n");
        }
    }
    
        
*3.main函数*

    int main(int argc,char const *argv[]){
        Mat_Graph G;
        createGraph(&G);
        int begin = 0;
        dijkstra(G,begin);
        return 0;
    }

*1.创建一个邻接表（邻接矩阵转换邻接表）,定义栈及其基本操作*

    #include<stdio.h>
    #include<stdlib.h>

    typedef int VertexType;
    typedef int ArcType;
    #define MAXVEX 100  

    typedef struct{
        VertexType vertex[MAXSIZE];
        ArcType arc[MAXSIZE][MAXSIZE];
        int vertex_num;
        int arc_num;
    }Mat_Graph;

    typedef struct ArcNode{
        int arc_vex;
        struct ArcNode *next;  //指向下一个邻接点
    }ArcNode;

    typedef struct VertexNode{
        int in;               //比之前所学邻接表多一个元素，用来记录其入度
        VertexType data;            
        ArcNode *head;    //指向边表的头指针
    }VertexNode;

    typedef VertexNode AdjList[MAXSIZE];

    typedef struct{
        AdjList adjList;  //邻接表    此处 AdjList adjList等价于VertexNode adjList[MAXVEX]
        int numVertexes;  
        int numArcs;      
    }GraphAdjList;

    typedef GraphAdjList* AdjList_Graph;          //为了创建二级指针，可以传入一个指针并可以对指针指向的地址进行修改

    int top = -1;
    int stack[MAXSIZE];

    void push(int e){
        if(top > MAXSIZE){
            printf("满了\n");
            return;
        }

        top++;
        stack[top] = e;    //入栈
    }

    int pop(){
        if(top == -1){
            printf("空的\n");
            return;
        }

        int elem = stack[top];
        top--;
        return elem;
    }

    int is_empty(){
        if(top == -1) return 0;
        else   return 1;
    }

    //先有邻接矩阵，再将其改为邻接表
    void createGraph(Mat_Graph *G){
        G->vertex_num = 6;
        G->arc_num = 8;

        for(int i = 0;i < G->vertex_num;i++){
            G->vertex[i] = i;    //前边改了VertexType为int
        }

        for(int i = 0;i < G->vertex_num;i++){
            for(int j = 0;j < G->vertex_num;j++){
                G->arc[i][j] = 0;
            }
        }

        G->arc[0][1] = 1;
        G->arc[0][3] = 1;
        G->arc[1][5] = 1;
        G->arc[2][1] = 1;
        G->arc[2][5] = 1;
        G->arc[4][0] = 1;
        G->arc[4][1] = 1;
        G->arc[4][5] = 1;
    }

    //邻接矩阵转换为邻接表
    void create_adjgraph(Mat_Graph G,AdjList_Graph* ALG){
        ArcNode* e;

        *ALG = (AdjList_Graph)malloc(sizeof(GraphAdjList));
        (*ALG)->vertex_num = G.vertex_num;
        (*ALG)->arc_num = G.arc_num;

        for(int i = 0;i < G.vertex_num;i++){
            (*ALG)->adjlist[i].in = 0;
            (*ALG)->adjlist[i].data = G.vertex[i];
            (*ALG)->adjlist[i].head = NULL;
        }

        for(int i = 0;i < G.vertex_num;i++){
            for(int j = 0;j < G->vertex_num;j++){
                if(G.arc[i][j] == 1){
                    e = (ArcNode*)malloc(sizeof(ArcNode));
                    e->arc_vex = j;
                    e->next = (*ALG)->adjlist[i].head;
                    (*ALG)->adjlist[i].head = e;
                    (*ALG)->adjlist[j].in++;        //对于j这个顶点，其入度加1
                }
            }
        }
    }

*2.拓扑排序*

    void topological_sort(AdjList_Graph ALG){
        ArcNode* e;
        int curr;
        int k;

        for(int i = 0;i < ALG->vertex_num;i++){
            if(ALG->adjlist[i].in == 0){
                push(i);
            }
        }

        while(is_empty() != 0){
            curr = pop();
            printf("V%d -> ",ALG->adjlist[curr].data);
            e = ALG->adjlist[curr].head;

            while(e != NULL){
                k = e->arc_vex;
                ALG->adjlist[k].in--;
                if(ALG->adjlist[k].in == 0){
                    push(k);
                }
                e = e->next;
            }
        }
    }

    int main(int argc,char const *argv[]){
        Mat_Graph G;
        AdjList_Graph ALG;
        create_graph(&G);
        create_adjgraph(G,&ALG);
        topological_sort(ALG);
        return 0;
    }
