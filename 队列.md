# 队列 queue
**先进先出** FIFO  
只允许在表的一端进行插入，而在另一端删除元素。  
允许插入的一端称为队尾（rear），允许删除的一端则称为队头（front）  
*保存的是队头元素的下标值，队尾元素下一个位置的下标值*
### 队列的顺序结构-初始化
~~~
#define MAXSIZE 100
typedef int ElemType;
typedef struct{
  ElemType data[MAXSIZE];
  int front;
  int rear;
}Queue;
void initQueue(Queue *Q){
  Q->front = 0;
  Q->rear = 0;
}
~~~
### 队列的顺序结构-出队
~~~
ElemType dequeue(Queue *Q){
  if(Q->front == Q->rear){
    printf("空的\n");
    return 0;
  }
  ElemType e = Q->data[Q->front];
  Q->front++;
  return e;
}

### 队列的顺序结构—调整队列
    int queueFull(Queue *Q)
    {
      if(Q->front > 0)
      {
        int step = Q->front;
        for(int i=Q->front;i <= Q->rear;++i)
        {
          Q->data[i - step] = Q->data[i];
        }
        Q->front = 0;
        Q->rear = Q->rear - step;
        return 1;
      }
      else
      {
        printf("真的满了\n");
        return 0;
      }
    }

### 队列的链式结构-初始化
    typedef struct QueueNode
    {
      ElemType data;
      struct QueueNode *next;
    }QueueNode;

    typedef struct
    {
      QueueNode *front;
      QueueNode *rear;
    }Queue;
    
   Queue* initQueue()  //仍然不理解
   {
     Queue *q = (Queue*)malloc(sizeof(Queue));
     QueueNode *node = (QueueNode*)malloc(sizeof(QueueNode));
     node->data = 0;
     node->next = NULL;
     q->front = node;
     q->rear = node;
     return q;
    }
### 队列的链式结构-入队
    //尾插法
    void equeue(Queue *q,ElemType e)
    {
      QueueNode *node = (QueueNode*)malloc(sizeof(QueueNode));
      node->data = e;
      node->next = NULL;
      q->rear->next = node;
      q->rear = node;
    }
### 队列的链式结构-出队
    int dequeue(Queue *q,ElemType *e)
    {
      QueueNode *node = q->front->next;
      *e = node->data;
      q->front->next = node->next;
      if(q->rear == node)
      {
        q->rear = q->front;
      }
      free(node);
      return 1;
    }
