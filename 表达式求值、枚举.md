# 枚举
将变量的值一一列举出来，变量的值只限于列举出来的值的范围内  
在枚举中列出的每一个值，称为枚举元素  
每一个枚举元素由系统定义了一个用序号表示的数值，从0开始，分别为0,1,2...  
语法:
~~~
enum 枚举名
{
   枚举元素......
}
~~~
    #include<stdio.h>
    enum weekday{
        mon,tue,wed,thu,fri,sat,sun
    };
    int main(){
        enum weekday a;
        a=mon;
        enum weekday b;
        b=tue;
        printf("%d\n",a);
        printf("%d\n",b);
        return 0;
    }  //输出0,1
    //若想从1开始计数，枚举定义时mon=1

# 后缀表达式求值
    #define MAXSIZE 100
    typedef int ElemType;

    typedef struct{
        ElemType *data;
        int top;
    }Stack;

    typedef enum{
        LEFT_PARE,RIGHT_PARE,
        ADD,SUB,MUL,DIV,MOD,
        EOS,NUM
    } contentType;  //左括号，右括号，加减乘除，求模，结束符，数字

    char expr[]="82/2+56*-";

    //初始化
    Stack* initStack(){
        Stack *s=(Stack*)malloc(sizeof(Stack));
        s->data=(ElemType*)malloc(sizeof(ElemType));
        s->top=-1;
        return s
    }
    
    //进栈/压栈
    int push(Stack *s,ElemType e)
    {
        if(s->top >= MAXSIZE-1){
            printf("满了\n");
            return 0;
        }
        s->top++;
        s->data[s->top] = e;  //拿top当下标
        return 1;
    }
    
    //出栈
    ElemType pop(Stack *s,ElemType *e)
    {
          if(s->top==-1){
              printf("空的\n");
              return 0;
         }
         *e = s->data[s->top];
         s->top--;
         return 1;
    }
    
    //获取栈顶元素
    int getTop(Stack *s, ElemType *e){
        if(s->top==-1){
            printf("空的\n");
            return 0;
        }
        *e = s->data[s->top];
        return 1;
    }

    contentType getToken(char *symbol,int *index){
        *symbol=expr[*index];  //按下标逐个取expr中的内容
        *index=*index+1;
        switch(*symbol){
            case '(':
                return LEFT_PARE;
            case ')':
                return RIGHT_PARE;
            case '+':
                return ADD;
            case '-':
                return SUB;
            case '*':
                return MUL;
            case '/':
                return DIV;
            case '%':
                return MOD;
            case '\0':
                return EOS;
            default:
                return NUM;
        }
    }

    int eval(Stack *s){
        char symbol;  //运算符
        int op1,op2;  //操作数1,2
        int index=0;  //索引，下标
        contentType token;
        token=getToken(&symbol,&index);
        ElemType result;
        while(token != EOS){
            if(token == NUM){
                push(s,symbol-'0');  //栈是ElemType类型（即int类型）的栈，但symbol是字符类型的数字，故symbol-'0'可得实际数字
            }
            else{
                pop(s,&op2);
                pop(s,&op1);

                switch(token){
                    case ADD:
                        push(s,op1+op2);
                        break;
                  case SUB:
                        push(s,op1-op2);
                        break;
                  case MUL:
                        push(s,op1*op2);
                        break;
                  case DIV:
                        push(s,op1/op2);
                        break;
                  case MOD:
                        push(s,op1%op2);
                        break;
                  default:
                        break;
                }
            }
            token=getToken(&symbol,&index);
        }
        pop(s,&result);
        printtf("%d\n",result);
        return 1;
    }

    int main(){
        Stack *s=initStack();
        eval(s);
        return 0;
    }
