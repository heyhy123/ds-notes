# 枚举
将变量的值一一列举出来，变量的值只限于列举出来的值的范围内  
在枚举中列出的每一个值，称为枚举元素  
每一个枚举元素由系统定义了一个用序号表示的数值，从0开始，分别为0,1,2...  
语法:
~~~
enum 枚举名
{
   枚举元素......
}
~~~
    #include<stdio.h>
    enum weekday{
        mon,tue,wed,thu,fri,sat,sun
    };
    int main(){
        enum weekday a;
        a=mon;
        enum weekday b;
        b=tue;
        printf("%d\n",a);
        printf("%d\n",b);
        return 0;
    }  //输出0,1
    //若想从1开始计数，枚举定义时mon=1

# 中缀表达式转后缀表达式
<img width="1734" height="906" alt="20fccc074faa4cd188168957f5db046e" src="https://github.com/user-attachments/assets/2088503f-c772-432c-aa46-80154fc2b3a5" />

中缀表达式：x/(i-j)*y ————> 后缀表达式：xij-/y *  
依旧利用栈  
**转化方法**：  
如果是操作数，直接输出   
如果是运算符，判断优先级，如果优先级大于栈顶元素，压入栈中，否则将栈顶元素出栈输出，然后将当前运算符压入栈中   
当左括号在栈外时，属于最高优先级，当左括号在栈里时，属于最低优先级  
如果是右括号，且栈顶元素不是左括号，持续出栈并输出，直到栈顶为左括号，出栈结束。最后将左括号也出栈，但不输出  
遇到结束符，将栈中元素依次输出 
<img width="1645" height="1011" alt="b410d233b190481051a6d67d98813e16" src="https://github.com/user-attachments/assets/1a19bce7-3cc8-4142-9271-22a60d4ca0cf" />

**转换过程**：  
先在栈中放入\0（优先级最低）  
x直接输出  
/（优先级为13）压入栈中  
(（压入栈后优先级为0）压入栈中  
i直接输出
-（优先级为12）压入栈中  
j直接输出  
) 开始出栈，故此时输出内容为xij-  
*（13）此时栈顶为/，故输出/，再将 *压入栈中  
y直接输出  
\0 将所有元素出栈  
故最终输出为：xij-/y * 

**代码**：  
~~~
#include<stdio.h>
#include<stblib.h>

#define MAXSIZE 100
typedef int ElemType;

typedef struct{
    ElemType *data;
    int top;
}Stack;

typedef enum{
    LEFT_PARE,RIGHT_PARE,
    ADD,SUB,MUL,DIV,MOD,
    EOS,NUM
} contentType;  //左括号，右括号，加减乘除，求模，结束符，数字

char expr[]="x/(i-j)*y";

//初始化
Stack* initStack(){
    Stack *s=(Stack*)malloc(sizeof(Stack));
    s->data=(ElemType*)malloc(sizeof(ElemType));
    s->top=-1;
    return s
}

//进栈/压栈
int push(Stack *s,ElemType e)
{
    if(s->top >= MAXSIZE-1){
        printf("满了\n");
        return 0;
    }
    s->top++;
    s->data[s->top] = e;  //拿top当下标
    return 1;
}

//出栈
ElemType pop(Stack *s,ElemType *e)
{
      if(s->top==-1){
          printf("空的\n");
          return 0;
     }
     *e = s->data[s->top];
     s->top--;
     return 1;
}

//获取栈顶元素
int getTop(Stack *s, ElemType *e){
    if(s->top==-1){
        printf("空的\n");
        return 0;
    }
    *e = s->data[s->top];
    return 1;
}

contentType getToken(char *symbol,int *index){
    *symbol=expr[*index];  //按下标逐个取expr中的内容
    *index=*index+1;
    switch(*symbol){
        case '(':
            return LEFT_PARE;
        case ')':
            return RIGHT_PARE;
        case '+':
            return ADD;
        case '-':
            return SUB;
        case '*':
            return MUL;
        case '/':
            return DIV;
        case '%':
            return MOD;
        case '\0':
            return EOS;
        default:
            return NUM;
    }
}

//token是ADD、SUB这些，所以要对其输出进行转化
int print_token(contentType token){
    switch(token){
              case ADD:
                    printf("+");
                    break;
              case SUB:
                    printf("-");
                    break;
              case MUL:
                    printf("*");
                    break;
              case DIV:
                    printf("/");
                    break;
              case MOD:
                    printf("%");
                    break;
              default:
                    return 0;
            }
    return 1;
}

//中缀表达式转后缀表达式
void postfix(Stack *s){
    int in_stack[] = {0,19,12,12,13,13,13,0};  //在栈内的优先级，根据数组的下标（数组下标与枚举中的数值对应），可匹配对应的运算符的优先级
    int out_stack[] = {20,19,12,12,13,13,13,0}; //在栈外的优先级
    contentType token;
    int index = 0;  //索引，下标
    s->top = 0;
    s->data[0] = EOS;
    char symbol;
    ElemType e;

    token=getToken(&symbol,&index);
    while(token != EOS){
        if(token == NUM){
            printf("%c",symbol);
        }
        else if(token == RIGHT_PARE){
            while(s->data[s->top] != LEFT_PARE){
                pop(s,&e);
                print_token(e);
            }
            pop(s,&e);
        }
        else{
            while(in_stack[s->data[s->top]] >= out_stack[token]){
                pop(s,&e);
                print_token(e);
            }
            push(s,token);
        }
        token=getToken(&symbol,&index);
    }

    pop(s,&e);
    token = e;
    while(token != EOS){
        print_token(token);
        pop(s,&e);
        token = e;
    }
    printf("\n");
}

int main(int argc,char const *argv[]){
    Stack *s=initStack();
    printf("%s\n",expr);
    postfix(s);
    return 0;
}
~~~
