## 折半插入排序

***代码如下***  
*binaryInsertSort函数*  

    void binaryInsertSort(int* data,int len){
        int low,high,mid;
        int key;

        //从第二个元素开始遍历，将其插入到前面已排好的子序列中
        for(int i = 1;i < len;i++){
            key = data[i];
            low = 0;
            high = i-1;

            while(low <= high){
                mid = (low + high)/2;
                if(data[mid] > key){
                    high = mid - 1;
                }
                else{
                    low = mid + 1;
                }                         //或者用 data[mid] > key ? high=mid-1 : low=mid+1;
            }

            //从后往前依次将大于key的元素后移，为key腾出位置
            int j = i-1;
            while(j >= low){
                data[j+1] = data[j];
                j--;
            }
            data[low] = key;
        }
    }

    void shellSort(int* data,int len){
        int temp;
        int step = len / 2;                 //初始化步长为数组长度的一半
        while(step >= 1){
            //对每个分组进行插入排序
            for(int i = step;i < len;i++){
                if(data[i] < data[i-step]){
                    temp = data[i];
                    int j = i - step;

                    //向前寻找合适插入位置，并逐步移动大于temp的元素
                    while(j >= 0 && temp < data[j]){
                        data[j+step] = data[j];
                        j = j - step;
                    }
                    data[j+step] = temp;
                }
            }
            step /= 2;
        }
    }

    #include <stdio.h>

    // 分区函数
    int partition(int* arr, int left, int right){
        int pivot = arr[left];              // 选最左元素为基准
        int i = left, j = right;
        while (i < j){
            // 右指针往左找比基准小的值
            while (i < j && arr[j] >= pivot){
                j--;
            }
            arr[i] = arr[j];
            // 左指针往右找比基准大的值
            while (i < j && arr[i] <= pivot){
                i++;
            }
            arr[j] = arr[i];
        }
    arr[i] = pivot;                       // 基准归位
    return i;
    }

    // 递归快速排序
    void quickSort(int* arr, int left, int right){
        if (left >= right)  return;
            
        int pivot = partition(arr, left, right);
        quickSort(arr, left, pivot - 1);
        quickSort(arr, pivot + 1, right);
    }

        int min;
        for(int i = 0;i < len;i++){
            min = i;
            for(int j = i;j < len;j++){
                if(data[min] > data[j]){
                    min = j;
                }
            }
            if(i != min){
                int temp = data[i];
                data[i] = data[min];
                data[min] = temp;
            }
        }
    }

    //向下调整堆（从 root 开始，len 为当前堆的大小）
    void adjust(int* data,int root,int len){
        int child;
        int temp = data[root];
        child = 2 * root;          //左孩子下标

        //只要存在孩子节点
        while(child <= len){
            //如果有右孩子，且比左孩子大，则选右孩子
            if(child < len && data[child] < data[child+1]){
                child++;
            }
            if(temp > data[child]){
                break;
            }
            data[child/2] = data[child];    //将子节点的值上移
            child = child * 2;
        }
        //把原 root 的值放到最终位置
        data[child/2] = temp;               //若temp被换了，则child此时是大于len的数，若temp没被换，则child/2就是原位置
    }
                
    void heapSort(int* data,int len){
        //1.建堆过程：从最后一个非叶子节点开始向上调整,结合二叉树性质，最后一个非叶子结点编号为[len / 2]
        for(int i = len/2;i > 0;i--){ 
            adjust(data,i len);
        }

        //2.排序过程：将堆顶最大值依次交换到数组末尾，并调整剩余部分为大根堆
        for(int i = len;i > 1;i--){
            swap(data,1,i);                 //将当前最大值（堆顶）移到末尾
            adjust(data,1,i-1);
        }
    }

    void merge(int* data,int left,int mid,int right){
        int temp[100];     //临时数组，用于存放合并结果
        int i = left;      //左半部分起始索引
        int j =mid + 1;    //右半部分起始索引
        int k = 0;         //temp数组下标

        //将两个有序部分按照大小关系依次放入 temp 中
        while(i <= mid && j <= right){
            if(data[i] <= data[j]){
                temp[k++] = data[i++];
            }
            else{
                temp[k++] = data[j++];
            }
        }
        //将剩余的左边部分拷贝到temp中（如果有）
        while(i <= mid){
            temp[k++] = data[i++];
        }
        //同理
        while(j <= right){
            temp[k++] = data[j++];
        }
        //将temp中的结果拷贝回原数组中，完成合并
        for(int t = 0;t < k;t++){
            data[left + t] = temp[t];
        }
    }

    void mergeSort(int* data,int left,int right){
        if(left < right){                                  //至少两个元素才需要排序
            int mid = (left + right) / 2;
            mergeSort(data,left,mid);
            mergeSort(data,mid+1,right);
            merge(data,left,mid,right);                    //将排好序的左右两部分合并
        }
    }
