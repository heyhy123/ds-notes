## 直接插入排序
<img width="284" height="194" alt="image" src="https://github.com/user-attachments/assets/128255e9-fd65-4c63-9d38-e11711c2413b" />

***代码如下***  
*1.insertSort函数*  

    void insertSort(int* data,int len){
        int key;            //用于暂存当前待插入的元素
        for(int i = 1;i < len;i++){    //从第二个元素开始逐个插入
            key = data[i];
            int j = i-1;    //从已排序序列的最后一个元素开始向前比较 
            
            while(j >=0 && data[j] > key){
                data[j+1] = data[j];
                j--;
            }

            data[j+1] = key;       //前边循环里有一个j--
        }
    }

*2.printArr函数*  

    void printArr(int arr[],int len){
        for(int i = 0;i < len;i++){
            printf("%d ",arr[i]);
        }
    }

*3.main函数*  

    int main(int argc,char const *argv[]){
        int data[] = {47,35,60,95,77};
        int len = sizeof(data) / sizeof(data[0]);

        insertSort(data,len);
        
        printArr(data,len);
        return 0;
    }

## 折半插入排序

***代码如下***  
*binaryInsertSort函数*  

    void binaryInsertSort(int* data,int len){
        int low,high,mid;
        int key;

        //从第二个元素开始遍历，将其插入到前面已排好的子序列中
        for(int i = 1;i < len;i++){
            key = data[i];
            low = 0;
            high = i-1;

            while(low <= high){
                mid = (low + high)/2;
                if(data[mid] > key){
                    high = mid - 1;
                }
                else{
                    low = mid + 1;
                }                         //或者用 data[mid] > key ? high=mid-1 : low=mid+1;
            }

            //从后往前依次将大于key的元素后移，为key腾出位置
            int j = i-1;
            while(j >= low){
                data[j+1] = data[j];
                j--;
            }
            data[low] = key;
        }
    }

    #include<stdio.h>
    void swap(int* data,int m,int n){
        int temp = data[m];
        data[m] = data[n];
        data[n] = temp;
    }
    
*2.bubbleSort函数*  

    void bubbleSort(int* data,int len){
        //外层循环实现趟数，共需len次比较（严格来说是 len-1 次）
        for(int i = 0;i < len;i++){
            //内层循环控制每趟中相邻元素的比较和交换，每次比较到 len-i-1
            for(int j = 0;j < len-i-1;j++){
                if(data[j] > data[j + 1]){
                    swap(data,j,j+1);
                }
            }
        }
    }

## 快速排序
<img width="288" height="170" alt="image" src="https://github.com/user-attachments/assets/822570b8-9bfa-44fa-83de-34e0ac772698" />
<img width="278" height="181" alt="image" src="https://github.com/user-attachments/assets/0f18c6f6-3613-49ad-988c-005476a2c29d" />

***代码如下***  
*quickSort函数*  

    #include <stdio.h>

    // 分区函数
    int partition(int arr[], int left, int right){
        int pivot = arr[left];              // 选最左元素为基准
        int i = left, j = right;
        while (i < j){
            // 右指针往左找比基准小的值
            while (i < j && arr[j] >= pivot)   j--;
            arr[i] = arr[j];
            // 左指针往右找比基准大的值
            while (i < j && arr[i] <= pivot)  i++;            
            arr[j] = arr[i];
        }
    arr[i] = pivot;                       // 基准归位
    return i;
    }

    // 递归快速排序
    void quickSort(int arr[], int left, int right){
        if (left >= right)  return;
            
        int mid = partition(arr, left, right);
        quickSort(arr, left, mid - 1);
        quickSort(arr, mid + 1, right);
    }
