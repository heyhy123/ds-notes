## 顺序查找
应用范围：  
1.顺序表或线性链表表示的静态查找表  
2.表内元素之间无序  

***代码如下***   
*1.search函数*

    #include<stdio.h>

    int search(int* data,int len,int value){
        for(int i = 0;i < len;i++){
            if(data[i] == value){
                return i;
            }
        }
        return -1;
    }

*2.main函数*

    int main(int argc,char const *argv[]){
        int arr[] = {15,54,76,6,9};
        int len = sizeof(arr) / sizeof(arr[0]);
        int pos = search(arr,len,54);
        printf("%d\n",pos);
        return 0;
    }

**改进：把待查关键字key存入表头（“哨兵”），从后向前逐个比较可免去查找过程中每一步都要检测是否查找完毕，加快速度。**  
***代码如下***  

    //在顺序表L中查找值为e的数据元素
    int search(SSTable ST,KeyType key){
        ST.R[0].key = key;                              //key存入顺序表表头
        for(i = ST.length;ST.R[i].key != key;--i);      //不用检测是否查找完毕，只用比较是否匹配
        return i;                                       //若成功返回其位置信息，否则返回0
    }

***代码如下***  
*1.binary_search函数*

    #include<stdio.h>

    int binary_search(int* data,int len,int value){
        int low = 0;
        int high = len-1;
        int mid;
        while(low <= high){
            mid = (low + high) / 2;
            if(value > data[mid]){
                low = mid + 1;
            }
            else if(value < data[mid]){
                high = mid - 1;
            }
            else{
                return mid;
            }
        }
        return 0;
    }

### 查找

***代码如下（和之前造二叉树的代码相似，但略有不同）***  
*1.造一颗二叉树*  

    #include<stdio.h>
    #include<stdlib.h>
    typedef int ElemType;     //定义二叉树节点数据类型，之前二叉树节点是字母，所以是char，此处为数字，故定义为int

    //定义二叉树节点结构体
    typedef struct TreeNode{          
        ElemType data;                //节点存储的数据（关键字）
        struct TreeNode *lchild;
        struct TreeNode *rchild;
    }TreeNode;

    typedef TreeNode* BiTree;         //定义二叉树指针类型，此定义含义：BiTree T等价于TreeNode *T,表示一个节点的指针

    //用数组模拟先序创建一棵二叉树（-1表示空节点）
    int treeArr[] = {70,55,49,30,-1,39,-1,-1,53,-1,-1,-1,80,75,-1,-1,98,95,-1,-1,-1};
    int idx = 0;

    //递归先序方式构建二叉树（含空节点）
    void createTree(BiTree *T){      //二级指针，若只使用一级指针即BiTree T,则无法改变T所指向的节点
        ElemType num;
        num = treeArr[idx++];        //取当前索引位置的值

        if(num == -1){               //-1代表空指针
            *T = NULL;
        }
        else{                        //创建新节点并递归构建左右子树
            *T = (BiTree)malloc(sizeof(TreeNode));
            (*T)->data = num;
            createTree(&(*T)->lchild);
            createTree(&(*T)->rchild);
        }
    }

    //先序遍历打印二叉树结点值（调式用）
    void preOrder(BiTree T){
        if(T == NULL){
            return;
        }
        printf("%d ",T->data);
        preOrder(T->lchild);
        preOrder(T->rchild);
    }

*2.查找函数*  

    //T:当前子树根节点   value:要查找的值    parent:当前节点的父节点    pos:指针变量，用于返回找到的节点
    int search_bst(BiTree T,int value,BiTree parent,BiTree *pos){
        if(T == NULL){
            *pos = parent;     //递归到空节点，查找失败，将父节点返回（后面内容可用于插入）
            return 0;
        }

        if(T->data == value){
            *pos = T;
            return 1;
        }

        if(T->data > value){
            return search_bst(T->lchild,value,T,pos);
        }
        else{
            return search_bst(T->rchild,value,T,pos);
        }
    }   

*3.main函数*  

    int main(int argc,char const *argv[]){
        BiTree T;
        createTree(&T);

        BiTree searchT;
        search_bst(T,53,NULL,&searchT);     //查找值为53的节点

        preOrder(T);       //输出树结构（调试用）
        printf("\n");

        printf("%d\n",seachT->data);     //输出查找到的节点值
        return 0;
    }

    int insert_bst(BiTree *T,int value){
        BiTree parent,pos;           //parent用于辅助search_bst, pos表示插入位置或已存在的位置
        BiTree curr;

        //查找插入的位置
        int status = search_bst(*T,value,NULL,&pos);
        if(status == 0){
            //如果未找到，创建新结点
            curr = (BiTree)malloc(sizeof(TreeNode));
            curr->data = value;
            curr->lchild = NULL;
            curr->rchild = NULL;
            
            //如果树为空，新结点作为根结点
            if(pos == NULL){
                *T = curr;
            }
            
            //插入为左子结点
            else if(value < pos->data){
                pos->lchild = curr;
            }
            
            //插入为右子结点
            else {
                pos->rchild = curr;
            }
            return 1;
        }
        else{               //已存在，不插入
            return 0;
        }
    }

*b站上的删除操作与老师讲的略有不同，老师讲的是为了不改变删除后二叉树的中序遍历的序列，用其直接前驱或直接后继替补；b站是从二叉树定义出发，用其左子树最大值或右子树最小值替补*  

***代码如下***  
*直接写删除的代码*  

    //真正执行删除的操作（分三种情况）
    int delete(BiTree *d){
        BiTree temp,record;
        //情况1：只有左子树或无子节点
        if((*d)->rchild == NULL){
            temp = *d;
            *d = (*d)->lchild;
            free(temp);
        }
        
        //情况2：只有右子树
        else if((*d)->lchild == NULL){
            temp = *d;
            *d = (*d)->rchild;
            free(temp);
        }

        //情况3：左右子树都有，找左子树中最大的节点替换
        else{
            temp == *d;
            record = (*d)->lchild;
            while(record->rchild !=NULL){
                temp = record;
                record = record->rchild;
            }
            (*d)->data = record->data;       //替换当前值
            
            if(temp != *d){                 //删除替代节点
                temp->rchild = record->lchild;
            }
            else{
                temp->lchild = record->lchild;
            }
            free(record);
        }
        return 1;
    }
            

    //递归查找并删除目标节点
    int delete_bst(BiTree *T,int value){
        if(*T == NULL){
            printf("not found!\n");
            return 0;
        }
        else{
            if((*T)->data == value){
                return delete(T);
            }
            else if((*T)->data > value){
                return delete_bst(&(*T)->lchild,value);
            }
            else{
                return delete_bst(&(*T)->rchild,value);
            }
        }
    }
